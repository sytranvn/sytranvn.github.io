Remove all tweets & retweets from your account

March 12, 2022 One-minute read

Open devtools console and run this script.

id = setInterval(() => {
    m = document.querySelector('[aria-label=More]')
    m.click()
    setTimeout(() => {
        try {
            const menuitem = document.querySelector('[role=menu]').querySelector('[role=menuitem]')
            if (menuitem?.textContent == 'Delete') { 
                menuitem.click()                               
                document.activeElement.click()
            }
        } catch(e) { 
            document.querySelector('[data-testid=unretweet]').click();
            document.querySelector('[data-testid="unretweetConfirm"]').click();
        }
    }, 300)
}, 300)

// wait until it's done and run
clearInterval(id)
